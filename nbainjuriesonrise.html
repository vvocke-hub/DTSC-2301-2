# Used Ai help
nba_injuries['Date of Injury'] = pd.to_datetime(nba_injuries['Date of Injury']) 

# Create year column
nba_injuries['Year'] = nba_injuries['Date of Injury'].dt.year

# Count injuries per year
injuries_by_year = nba_injuries['Year'].value_counts().sort_index()

# Plot bar graph
plt.figure(figsize=(12,6))
plt.bar(injuries_by_year.index, injuries_by_year.values)
plt.title('NBA Injuries by Year')
plt.xlabel('Year')
plt.ylabel('Number of Injuries')
plt.show()
